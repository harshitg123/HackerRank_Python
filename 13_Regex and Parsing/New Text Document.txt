import re
for i in range(int(input())):
    uids="".join(sorted(input()))
    print(uids)
    if len(uids)==10:
        uid=re.search(r"([0-9]{3,})(?!.*\1)*([A-Z]{2,})(?!.*\1)*$",uids)
        print(uid.group())
    else:
        print("Invalid")
    if uid=="True":
        print("Valid")
    #else:
    #    print("Invalid")
